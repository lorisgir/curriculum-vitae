\sectionTitle{Capacità}{\faCogs}
\begin{keywords}
	\keywordsentry{Capacità organizzative}{Capacità di lavorare in situazioni di stress, legate soprattutto al rapporto con il pubblico e alle scadenze fiscali delle attività lavorative.}
	\keywordsentry{Capacità Interpersonali}{Capacità nella gestione del lavoro di gruppo e nella realizzazione degli obbiettivi
		individuali:
		\begin{itemize}
			\item Disponibilità ad affrontare e confrontare nuove possibilità.
			\item Buone capacità comunicative e relazionali sviluppate sia in ambito scolastico in occasione di progetti di gruppo, sia lavorativo.
			\item Capacità di gestire autonomamente i propri impegni con puntualità e precisione.
		\end{itemize}
	}
	
\end{keywords}

\sectionTitle{Capacità}{\faCogs}
\begin{keywords}
	\keywordsentry{Capacità organizzative}{Capacità di lavorare in situazioni di stress, legate soprattutto al rapporto con il pubblico e alle scadenze fiscali delle attività lavorative.}
	\keywordsentry{Capacità interpersonali}{Capacità nella gestione del lavoro di gruppo e nella realizzazione degli obiettivi
		individuali:
		\begin{itemize}
			\item Disponibilità ad affrontare e confrontare nuove possibilità.
			\item Buone capacità comunicative e relazionali.
			\item Capacità di gestire autonomamente i propri impegni con puntualità e precisione.
		\end{itemize}
	}
	\keywordsentry{Competenze imprenditoriali}{Esperienza diretta nella creazione e gestione di una startup tecnologica:
		\begin{itemize}
			\item Abitudine a prendere decisioni rapide in contesti incerti e in continua evoluzione.
			\item Attitudine al problem solving e alla sperimentazione continua, anche con risorse limitate.
			\item Esperienza nella definizione e validazione di un prodotto, dalla progettazione allo sviluppo e rilascio.
			\item Gestione trasversale di attività non strettamente tecniche: comunicazione con stakeholder, pianificazione, fundraising, marketing e amministrazione.
		\end{itemize}
	}
\end{keywords}
